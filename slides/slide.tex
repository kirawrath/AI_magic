\section{The Approach}
\frame{\frametitle{Genetic Algorithm}
\begin{itemize}
	\item Create a set of X random decks (generation)
	\item Simulate a fight between each other using simple heuristics
	\item Compute scores (win/lose)
	\item Select the best decks from the current generation (won - lost), ratio $> 0.6$
	\item Reproduce (crossover)
	\item Mutate
\end{itemize} 
}
